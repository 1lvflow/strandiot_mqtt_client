## 🔌 硬件接线

### 烧录模式接线

使用 USB 转 TTL 模块烧录程序时：

| ESP-01S | → | USB 转 TTL | 说明 |
|---------|---|------------|------|
| VCC | → | 3.3V | 供电 |
| GND | → | GND | 共地 |
| TX | → | RX | 交叉连接 |
| RX | → | TX | 交叉连接 |
| **GPIO0** | → | **GND** | ⚠️ 关键！进入烧录模式 |
| EN (CH_PD) | → | 3.3V | 使能引脚 |
| GPIO2 | → | 悬空 | 不需要连接 |
| RST | → | 悬空 | 烧录卡顿时短接 GND |

### 正常运行接线

烧录完成后：

1. **断开电源**
2. **拔掉 GPIO0 → GND 的连接线**
3. 将 ESP-01S **直接插入**继电器底板的排母
4. 用 **5V USB 电源**给继电器底板供电

### 强电接线（220V）

| 继电器接线柱 | 连接 | 说明 |
|-------------|------|------|
| **COM** | 火线进线（L） | 220V 火线输入 |
| **NO** | 灯的火线 | 灯的另一端接零线（N） |
| **NC** | 不接 | 常闭端，一般不用 |

> ⚠️ **安全警告**：接线前务必断电！不懂强电的请找电工帮忙。

## 📝 完整代码

### 主程序代码

```cpp
#include <ESP8266WiFi.h>
#include <PubSubClient.h>

// ========== 请修改以下配置 ==========
const char* ssid = "你的WiFi名称";
const char* password = "你的WiFi密码";

const char* mqtt_server = "192.168.1.100";  // MQTT 服务器 IP
const int mqtt_port = 1883;
const char* mqtt_user = "";      // 无认证则留空
const char* mqtt_password = "";  // 无认证则留空

// MQTT 主题
const char* light_command_topic = "home/light/switch";   // 接收命令
const char* light_state_topic = "home/light/state";      // 上报状态
// ==================================

#define RELAY_PIN 0    // 继电器控制引脚（GPIO0）

WiFiClient espClient;
PubSubClient client(espClient);
bool relayState = HIGH;  // HIGH=断开，LOW=吸合

void setup_wifi() {
delay(10);
Serial.println();
Serial.print("正在连接 WiFi: ");
Serial.println(ssid);

WiFi.begin(ssid, password);

while (WiFi.status() != WL_CONNECTED) {
 delay(500);
 Serial.print(".");
}

Serial.println();
Serial.println("WiFi 已连接");
Serial.print("IP 地址: ");
Serial.println(WiFi.localIP());
}

void callback(char* topic, byte* payload, unsigned int length) {
String message = "";
for (int i = 0; i < length; i++) {
 message += (char)payload[i];
}

if (String(topic) == light_command_topic) {
 if (message == "ON") {
   digitalWrite(RELAY_PIN, LOW);
   relayState = LOW;
   publishState();
 } 
 else if (message == "OFF") {
   digitalWrite(RELAY_PIN, HIGH);
   relayState = HIGH;
   publishState();
 }
}
}

void publishState() {
if (relayState == LOW) {
 client.publish(light_state_topic, "ON");
} else {
 client.publish(light_state_topic, "OFF");
}
}

void reconnect() {
while (!client.connected()) {
 String clientId = "ESP01S-" + String(random(0xffff), HEX);
 
 if (client.connect(clientId.c_str(), mqtt_user, mqtt_password)) {
   client.subscribe(light_command_topic);
   publishState();
 } else {
   delay(5000);
 }
}
}

void setup() {
Serial.begin(115200);
pinMode(RELAY_PIN, OUTPUT);
digitalWrite(RELAY_PIN, HIGH);

setup_wifi();
client.setServer(mqtt_server, mqtt_port);
client.setCallback(callback);
}

void loop() {
if (!client.connected()) {
 reconnect();
}
client.loop();
}
